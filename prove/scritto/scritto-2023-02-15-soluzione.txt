monitor redblack {
    int last = -1;
    double sum[2]
    int count[2]
    condition ok2col[2]

    double rb(int color, double value) {
        if (color == last)
            ok2col[color].wait()
        last = color;
        sum[color] += value;
        count[color]++;
        mean = sum[color] / count[color]
        ok2col[1-color].signal()
        return mean
    }
}
